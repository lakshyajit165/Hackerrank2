void quick_sort(int A[],int l,int u)
{
    int j;
    if(l<u){
        j=partition(A,l,u);
        quick_sort(A,l,j-1);
        quick_sort(A,j+1,u);
    }
}
int partition(int A[],int l,int u){
    int v,i,j,temp;
    v=A[l];
    i=l;
    j=u+1;
    do{
        do
            i++;
        while(A[i]<v&&i<=u);
        do
            j--;
        while(v<A[j]);
        if(i<j){
            temp=A[i];
            A[i]=A[j];
            A[j]=temp;
        }
    }while(i<j);
    A[l]=A[j];
    A[j]=v;
    return(j);
}ß
